async function connectArduino() { try { const port = await navigator.serial.requestPort(); await port.open({ baudRate: 9600 }); const decoder = new TextDecoderStream(); const inputClosed = port.readable.pipeTo(decoder.writable); const reader = decoder.readable.getReader(); console.log("Порт открыт, слушаю Arduino..."); while (true) { const { value, done } = await reader.read(); if (done) break; // value — это строка из Arduino, например "GOTO:2" console.log("Пришло от Arduino:", value); // Ищем цифру в строке const match = value.match(/GOTO:(\d)/); if (match) { const floor = parseInt(match[1]); console.log("Едем на этаж:", floor); callElevator(floor); } } } catch (e) { alert("Ошибка Serial: " + e.message); } }
